
CREATE FUNCTION DEMO_FUNC ( IN e_type INT, IN p2 VARCHAR(255) )
RETURNS INT
BEGIN ATOMIC
DECLARE max_count INT;
SELECT COUNT(*) FROM p2 INTO max_count;
RETURN max_count;
END;
-- :

CREATE PROCEDURE DEMO_PROC ( IN e_type INT, OUT p2 INT )
MODIFIES SQL DATA
BEGIN ATOMIC
SET p2 = e_type;
END;
-- :
